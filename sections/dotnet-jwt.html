TokenService — generazione JWTC#
// Services/TokenService.cs
public class TokenService {
    private readonly IConfiguration _cfg;
    public TokenService(IConfiguration cfg) => _cfg = cfg;

    public string Generate(User user) {
        var key  = new SymmetricSecurityKey(
            Encoding.UTF8.GetBytes(_cfg["Jwt:Key"]!));
        var cred = new SigningCredentials(key, SecurityAlgorithms.HmacSha256);

        var claims = new[] {
            new Claim(ClaimTypes.NameIdentifier, user.Id.ToString()),
            new Claim(ClaimTypes.Email, user.Email),
            new Claim(ClaimTypes.Role, user.Role)
        };

        var token = new JwtSecurityToken(
            expires: DateTime.UtcNow.AddHours(8),
            signingCredentials: cred,
            claims: claims);

        return new JwtSecurityTokenHandler().WriteToken(token);
    }
}

⚠️ La chiave JWT deve avere almeno 32 caratteri (256 bit per HMAC-SHA256). Una chiave troppo corta causa un'eccezione a runtime. Non usare mai la stessa chiave in sviluppo e produzione.

Login endpointPOST
[AllowAnonymous]
[HttpPost("login")]
public async Task<IActionResult> Login([FromBody] LoginDto dto) {
    var user = await _userRepo.FindByEmailAsync(dto.Email);

    // Messaggio generico: non rivelare se è email o password errata
    if (user is null || !BCrypt.Verify(dto.Password, user.PasswordHash))
        return Unauthorized();

    var token = _tokenService.Generate(user);
    return Ok(new { token });
}

// DTO
public record LoginDto(string Email, string Password);
Claims disponibili lato controller
// Legge l'ID utente dal token
var userId = User.FindFirstValue(ClaimTypes.NameIdentifier);

// Legge email
var email = User.FindFirstValue(ClaimTypes.Email);

// Legge ruolo (usato con [Authorize(Roles = "Admin")])
var role = User.FindFirstValue(ClaimTypes.Role);

I claims sono disponibili tramite User (ClaimsPrincipal) in qualsiasi controller con [Authorize]. Il token viene decodificato automaticamente dal middleware JWT — nessun codice aggiuntivo richiesto.

Hashing password con BCrypt
# Installa pacchetto
dotnet add MyProject.Api package BCrypt.Net-Next

// Hash alla registrazione utente
user.PasswordHash = BCrypt.HashPassword(plainTextPassword);

// Verifica al login
bool valid = BCrypt.Verify(plainTextPassword, user.PasswordHash);

BCrypt include il salt nell'hash — non serve memorizzarlo separatamente. Verify() lo estrae e confronta internamente.
