Modal — add/edit Mode
Come discriminare il comportamento del modal in base al contesto di apertura
data.ts — selectedItem + modalMode component 
export class Data {
  showModal  = false;
  selectedItem: DataItem | null = null;
  modalMode: 'add' | 'edit' = 'add';

  addItem() {
    this.selectedItem = null;         // modal vuoto
    this.modalMode   = 'add';
    this.showModal   = true;
  }

  editItem(item: DataItem) {
    this.selectedItem = item;           // item da modificare
    this.modalMode   = 'edit';
    this.showModal   = true;
  }

  saveItem() {
    if (this.modalMode === 'add') {
      // dispatch azione di creazione
    } else {
      // dispatch azione di aggiornamento
    }
    this.showModal = false;
  }
}
click Edit(item) → editItem(item) → selectedItem = item, modalMode = 'edit', showModal = true
↓
modal → mostra selectedItem nei campi
↓
utente clicca Salva → confirm() → saveItem() → if modalMode === 'edit' → dispatch update
📌 Principio
Il modal emette (confirmed). Il component parent (data.ts) decide cosa fare. Il modal rimane generico e riutilizzabile.
