Service
Responsabile delle chiamate HTTP. Ritorna sempre un Observable.
product.service.ts service 
// product.service.ts
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';
import { IProductModel } from '../interface/product.interface';

@Injectable({ providedIn: 'root' })
export class ProductService {
  private baseUrl = '/api/products';

  constructor(private http: HttpClient) {}

  getProducts(params?: any): Observable<IProductModel> {
    return this.http.get<IProductModel>(this.baseUrl, { params });
  }
}
📌 Regola
Il Service non conosce NGXS. Non dispatcha nulla. Non tocca lo store. Restituisce solo un Observable.
