1 — Model e Actionsaction
// user.model.ts
export interface User {
  id: string;
  name: string;
  email: string;
}

export interface UserStateModel {
  users: User[];
  loading: boolean;
  lastSearchTerm: string | null;
  error: string | null;
}

// user.actions.ts
export class SearchUsers {
  static readonly type = '[Users] Search';
  constructor(public searchTerm: string) {}
}

export class SearchUsersSuccess {
  static readonly type = '[Users] Search Success';
  constructor(public users: User[], public searchTerm: string) {}
}

export class SearchUsersFailure {
  static readonly type = '[Users] Search Failure';
  constructor(public error: string, public searchTerm: string) {}
}
2 — Service HTTPservice
@Injectable({ providedIn: 'root' })
export class UserService {
  private apiUrl = 'https://localhost:5001/api/users';

  constructor(private http: HttpClient) {}

  searchUsers(term: string): Observable<User[]> {
    // GET /api/users?search=term
    return this.http.get<User[]>(`${this.apiUrl}?search=${term}`);
  }
}
3 — State completostate
@State<UserStateModel>({
  name: 'users',
  defaults: { users: [], loading: false, lastSearchTerm: null, error: null }
})
@Injectable()
export class UserState {

  constructor(private userService: UserService) {}

  @Selector() static getUsers(s: UserStateModel) { return s.users; }
  @Selector() static isLoading(s: UserStateModel) { return s.loading; }
  @Selector() static getLastSearchTerm(s: UserStateModel) { return s.lastSearchTerm; }
  @Selector() static getError(s: UserStateModel) { return s.error; }

  @Action(SearchUsers)
  searchUsers(ctx: StateContext<UserStateModel>, action: SearchUsers) {
    const term = action.searchTerm;
    ctx.patchState({ loading: true, error: null, lastSearchTerm: term });

    return this.userService.searchUsers(term).pipe(
      tap((users: User[]) => {
        ctx.dispatch(new SearchUsersSuccess(users, term));
      }),
      catchError(err => {
        return of(ctx.dispatch(new SearchUsersFailure(err.message, term)));
      })
    );
  }

  @Action(SearchUsersSuccess)
  searchSuccess(ctx: StateContext<UserStateModel>, action: SearchUsersSuccess) {
    ctx.patchState({ users: action.users, loading: false, error: null });
  }

  @Action(SearchUsersFailure)
  searchFailure(ctx: StateContext<UserStateModel>, action: SearchUsersFailure) {
    ctx.patchState({ users: [], loading: false, error: action.error });
  }
}
4 — Component e Templatecomponent
@Component({
  selector: 'app-user-search',
  template: `
    <input #searchInput
           (keyup.enter)="onSearch(searchInput.value)"
           placeholder="Cerca per nome o email" />
    <button (click)="onSearch(searchInput.value)">Cerca</button>

    <div *ngIf="loading$ | async">⏳ Caricamento...</div>

    <div *ngIf="error$ | async as error" style="color:red">
      ❌ Errore: {{ error }}
    </div>

    <ul *ngIf="(users$ | async) as users">
      <li *ngFor="let user of users">
        <strong>{{ user.name }}</strong> ({{ user.email }})
      </li>
    </ul>

    <div *ngIf="(users$ | async)?.length === 0
                && !(loading$ | async)
                && (lastSearchTerm$ | async)">
      😕 Nessun utente trovato per "{{ lastSearchTerm$ | async }}"
    </div>
  `
})
export class UserSearchComponent {
  users$          = this.store.select(UserState.getUsers);
  loading$        = this.store.select(UserState.isLoading);
  error$          = this.store.select(UserState.getError);
  lastSearchTerm$ = this.store.select(UserState.getLastSearchTerm);

  constructor(private store: Store) {}

  onSearch(term: string) {
    if (term.trim()) {
      this.store.dispatch(new SearchUsers(term));
    }
  }
}
Flusso passo per passowalkthrough
1. Utente digita "Mario" e preme Invio → onSearch('Mario')
2. Component dispatcha → new SearchUsers('Mario')
3. State riceve SearchUsers:
   - patchState loading=true, lastSearchTerm='Mario'
   - UI mostra spinner, ricorda il termine
   - Chiama userService.searchUsers('Mario') → HTTP GET

4. Backend .NET risponde:
   [{ id:'1', name:'Mario Rossi', email:'mario@example.com' }]

5. State: tap dispatcha SearchUsersSuccess(users, 'Mario')
6. SearchUsersSuccess: patchState users=[...], loading=false
7. Template:
   - loading$ → false → spinner scompare
   - users$ emette il nuovo array → *ngFor aggiorna la lista

8. Se errore (500 backend):
   - SearchUsersFailure aggiorna error
   - UI mostra messaggio rosso
