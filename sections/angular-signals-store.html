1 — Il Modello (Interfaces)model

Definiamo la gerarchia dei dati. I tipi viaggiano identici tra frontend e backend. 

// vendita.model.ts
export interface Sal {
  id: number;
  importo: number;
  date: string;
}

export interface Ordine {
  id: number;
  number: string;
  importo: number;
  listaSal: Sal[];
}

export interface Offerta {
  id: number;
  number: string;
  stato: number;
  ordini: Ordine[];
}
2 — Lo Store / Serviceservice

Tre sezioni distinte dentro un unico @Injectable: STATE (signal privati scrivibili), SELECTORS (signal pubblici in sola lettura) e ACTIONS (metodi che modificano lo stato). 

// vendita.store.ts
import { Injectable, signal, computed, inject } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { tap } from 'rxjs';
import { Offerta } from './vendita.model';

@Injectable({ providedIn: 'root' })
export class VenditaStore {
  private http = inject(HttpClient);

  // --- STATE (signal privati scrivibili) ---
  private _offerte = signal<Offerta[]>([]);
  private _loading = signal<boolean>(false);
  private _error   = signal<string | null>(null);

  // --- SELECTORS (pubblici, in sola lettura) ---
  readonly offerte = this._offerte.asReadonly();
  readonly loading = this._loading.asReadonly();
  readonly error   = this._error.asReadonly();

  // --- COMPUTED (stato derivato, ricalcolo automatico) ---
  readonly totaleProgetto = computed(() =>
    this._offerte().reduce((acc, off) => acc + (off.ordini[0]?.importo || 0), 0)
  );

  // --- ACTIONS ---
  loadOfferte(jobId: string) {
    this._loading.set(true);
    this._error.set(null);

    this.http.get<Offerta[]>(`/api/jobs/${jobId}/vendite`)
      .pipe(tap(data => this._offerte.set(data)))
      .subscribe({
        error:    (err) => this._error.set('Errore nel caricamento dati'),
        complete: ()    => this._loading.set(false)
      });
  }

  updateOffertaStatus(id: number, nuovoStato: number) {
    this._offerte.update(list =>
      list.map(o => o.id === id ? { ...o, stato: nuovoStato } : o)
    );
  }
}
asReadonly() — perché usarlo
Esporre il signal con asReadonly() impedisce al componente di chiamare .set() o .update() direttamente, forzando l'uso dei metodi action. È l'equivalente Signals di un getter NGXS.
3 — Il Componentecomponent

Il componente è snello: inietta lo store, chiama l'azione in ngOnInit e legge i segnali nel template. 

// vendita.component.ts
import { Component, OnInit, inject } from '@angular/core';
import { CommonModule } from '@angular/common';
import { VenditaStore } from './vendita.store';

@Component({
  selector: 'app-vendita-page',
  standalone: true,
  imports: [CommonModule],
  template: `
    <div class="container mt-4">
      <h2>Monitoraggio Vendite</h2>

      <div *ngIf="store.loading()" class="spinner-border text-danger"></div>
      <div *ngIf="store.error()" class="alert alert-danger">{{ store.error() }}</div>

      <div class="card mb-3 bg-dark text-white" *ngIf="!store.loading()">
        <div class="card-body">
          Totale Commessa: {{ store.totaleProgetto() | currency:'EUR' }}
        </div>
      </div>

      <table class="table table-bordered align-middle">
        <thead class="table-light">
          <tr>
            <th>Offerta</th><th>SAL / Stato</th><th>Azioni</th>
          </tr>
        </thead>
        <tbody>
          <tr *ngFor="let off of store.offerte()">
            <td><strong>{{ off.number }}</strong></td>
            <td>
              <div *ngFor="let sal of off.ordini[0]?.listaSal" class="small border-bottom">
                SAL del {{ sal.date }}: {{ sal.importo | currency:'EUR' }}
              </div>
            </td>
            <td>
              <button class="btn btn-sm btn-outline-primary"
                      (click)="store.updateOffertaStatus(off.id, 2)">
                Accetta
              </button>
            </td>
          </tr>
        </tbody>
      </table>
    </div>
  `
})
export class VenditaComponent implements OnInit {
  readonly store = inject(VenditaStore);

  ngOnInit() {
    this.store.loadOfferte('JOB-2024-001');
  }
}
Signals vs NGXS — Confronto Praticoconfronto
Aspetto NGXS Signals
Chiamare un'azione store.dispatch(new LoadOfferte(id)) store.loadOfferte(id)
Leggere lo stato nel template offerte$ | async store.offerte() — le parentesi leggono il valore corrente
Stato derivato Selector con createSelector computed(() => ...) — ricalcolo automatico
Aggiornamento DOM Zone.js rileva ogni cambiamento Angular aggiorna solo il nodo che dipende dal signal cambiato
File necessari actions.ts · state.ts · selectors.ts Un solo vendita.store.ts
Le parentesi contano
store.offerte è il signal (oggetto). store.offerte() è il valore attuale (array). Nel template Angular devi sempre usare le parentesi per "leggere" il segnale; senza parentesi ottieni l'oggetto signal, non i dati.
