1 — Signal Base signal

Creazione e operazioni

import { signal, WritableSignal, Signal } from '@angular/core';

// WritableSignal: può essere modificato
const count: WritableSignal<number> = signal(0);

count.set(5);                        // Imposta valore
count.update(n => n + 1);           // Aggiorna basandosi sul valore precedente
count.mutate(arr => arr.push(4));  // Solo per array/oggetti

// Signal readonly: solo lettura (di solito da computed)
const doubled: Signal<number> = computed(() => count() * 2);
// doubled.set(10); ❌ Errore: readonly

Writable vs Readonly

export class CounterService {
  private _count = signal(0);

  // Public readonly - esponi solo lettura
  readonly count = this._count.asReadonly();

  increment() {
    this._count.update(n => n + 1);
  }
}

export class MyComponent {
  service = inject(CounterService);

  ngOnInit() {
    console.log(this.service.count()); // ✅ Leggi
    // this.service.count.set(5); ❌ Errore: readonly
    this.service.increment();           // ✅ Modifica tramite metodo
  }
}
2 — Computed (valori derivati) computed

I computed sono lazy: si ricalcolano solo quando vengono letti e solo se le dipendenze sono cambiate. 

// Computed: valori derivati automatici
filteredProducts = computed(() => {
  const search = this.searchTerm();
  const min    = this.minPrice();
  return this.products()
    .filter(p => p.name.includes(search))
    .filter(p => p.price >= min);
});

// I computed sono LAZY: si ricalcolano solo quando vengono letti
totalPrice = computed(() => {
  console.log('Ricalcolo total'); // Vedi quando si ricalcola
  return this.items().reduce((sum, i) => sum + i.price, 0);
});
3 — Effect + untracked() effect

Effect standard

constructor() {
  effect(() => {
    console.log('Count:', this.count());
    // Si esegue quando count cambia
  });
}

untracked() — leggere senza creare dipendenza 

import { effect, untracked } from '@angular/core';

export class LogComponent {
  count    = signal(0);
  userName = signal('Paolo');

  constructor() {
    effect(() => {
      // count crea dipendenza → effect si ri-esegue quando cambia
      const currentCount = this.count();

      // userName NON crea dipendenza → effect NON si ri-esegue quando cambia
      const name = untracked(() => this.userName());

      console.log(`${name}: count è ${currentCount}`);
    });
  }
  // count.set(1)         → effect si esegue  ✅
  // userName.set('Mario') → effect NON si esegue ✅
}
Quando usare untracked
Logging con context info che non dovrebbe triggerare l'effect · Accedere a configurazione statica dentro effect · Evitare dipendenze circolari
4 — Interoperabilità con RxJS rxjs-interop

toSignal() — Observable → Signal

import { toSignal } from '@angular/core/rxjs-interop';
import { interval } from 'rxjs';

export class TimerComponent {
  private http = inject(HttpClient);

  // initialValue: valore iniziale mentre carica (signal type: T)
  users = toSignal(
    this.http.get<User[]>('/api/users'),
    { initialValue: [] as User[] }
  );

  // Con gestione errori
  usersOrError = toSignal(
    this.http.get<User[]>('/api/users').pipe(catchError(() => of([]))),
    { initialValue: [] }
  );

  // Timer reattivo
  seconds = toSignal(interval(1000), { initialValue: 0 });

  // requireSync: signal type T (no undefined) — errore se stream non emette subito
  config = toSignal(this.http.get<Config>('/api/config'), { requireSync: true });

  // Nessuna opzione → signal type: T | undefined
}

toObservable() — Signal → Observable 

import { toObservable } from '@angular/core/rxjs-interop';
import { debounceTime, switchMap } from 'rxjs';

export class SearchComponent {
  private http = inject(HttpClient);
  searchTerm = signal('');

  // Converti signal in Observable per usare operatori RxJS
  searchResults$ = toObservable(this.searchTerm).pipe(
    debounceTime(300),
    switchMap(term => this.http.get(`/api/search?q=${term}`))
  );

  // Poi riconverti in signal
  results = toSignal(this.searchResults$, { initialValue: [] });
}
Pattern: Signal → Observable → RxJS → Signal 
export class AutocompleteComponent {
  query = signal('');

  suggestions = toSignal(
    toObservable(this.query).pipe(
      debounceTime(300),
      distinctUntilChanged(),
      switchMap(q => q ? this.http.get(`/api/suggest?q=${q}`) : of([]))
    ),
    { initialValue: [] }
  );
}
5 — Signal Inputs/Outputs (Angular 17.1+) component

Signal Inputs — passare dati al componente

import { input, InputSignal } from '@angular/core';

export class UserCardComponent {
  userName = input<string>('');         // InputSignal<string>
  age      = input<number>(0);
  userId   = input.required<number>(); // required — InputSignal<number>

  // Input con transform
  disabled = input(false, {
    transform: (value: boolean | string) =>
      typeof value === 'string' ? value === '' : value
  });

  // Computed basato su input — reattivo automaticamente
  displayName = computed(() => `${this.userName()} (${this.age()} anni)`);

  constructor() {
    effect(() => console.log('User:', this.userName(), this.age()));
  }
}
// Nel parent: <app-user-card [userName]="'Paolo'" [age]="35" [userId]="123" />
Vantaggi Signal Inputs
Reattività automatica (no ngOnChanges) · Type-safe · Funzionano con computed/effect · Immutabili (readonly)

Signal Outputs — emettere eventi 

import { output } from '@angular/core';

export class CounterComponent {
  count        = signal(0);
  countChanged = output<number>(); // OutputEmitterRef<number>

  increment() {
    this.count.update(n => n + 1);
    this.countChanged.emit(this.count());
  }
}
// Nel parent: <app-counter (countChanged)="onCountChanged($event)" />

Pattern 3: Effect per chiamate reattive 

export class FilteredListComponent {
  private service = inject(ProductService);
  categoryId = signal(1);
  searchTerm = signal('');
  products   = signal<Product[]>([]);
  loading    = signal(false);

  constructor() {
    // Ricarica automaticamente quando cambiano i filtri
    effect(() => {
      const cat    = this.categoryId();
      const search = this.searchTerm();
      this.loading.set(true);
      this.service.search(cat, search).subscribe({
        next: (data) => { this.products.set(data); this.loading.set(false); }
      });
    });
  }
}

}
7 — Best Practices best practices

✅ Naming conventions

// Signal: nomi semplici
count     = signal(0);
userName  = signal('');
isLoading = signal(false);

// Computed: nomi descrittivi
filteredItems = computed(() => ...);
totalPrice    = computed(() => ...);
isFormValid   = computed(() => ...);

// Observable (se usi RxJS): suffisso $
users$ = this.http.get<User[]>(...);

✅ Incapsulamento in services 

@Injectable({ providedIn: 'root' })
export class CartService {
  private http = inject(HttpClient);

  private _items   = signal<CartItem[]>([]);
  private _loading = signal(false);

  readonly items   = this._items.asReadonly();
  readonly loading = this._loading.asReadonly();
  readonly total   = computed(() =>
    this._items().reduce((sum, item) => sum + item.price, 0)
  );

  addItem(item: CartItem) {
    this._items.update(items => [...items, item]);
  }

  loadCart() {
    this._loading.set(true);
    this.http.get<CartItem[]>('/api/cart').subscribe({
      next: (data) => { this._items.set(data); this._loading.set(false); }
    });
  }
}

✅ Evita effetti collaterali nei computed 

// ❌ MALE: side-effect in computed
badComputed = computed(() => {
  const value = this.count();
  console.log('Computing...');        // ❌
  localStorage.setItem('count', value); // ❌
  return value * 2;
});

// ✅ BENE: computed puro + side-effect in effect
goodComputed = computed(() => this.count() * 2);

constructor() {
  effect(() => {
    const value = this.count();
    console.log('Count:', value);           // ✅
    localStorage.setItem('count', value); // ✅
  });
}

✅ Cleanup in effect quando necessario 

constructor() {
  effect((onCleanup) => {
    const shouldPoll = this.autoPoll();
    if (shouldPoll) {
      const interval = setInterval(() => this.fetchData(), 5000);
      onCleanup(() => clearInterval(interval)); // ✅ Cleanup
    }
  });
}
Signal per stato locale, Service per condivisione
Component: formData = signal(...) — solo per questo component.
Service @Injectable({ providedIn: 'root' }): private _user = signal<User | null>(null) — condiviso tra tutti i component.

Prima (senza signal)

export class OldComponent {
  products:  Product[] = [];
  loading    = false;
  searchTerm = '';

  get filteredProducts() {
    return this.products
      .filter(p => p.name.includes(this.searchTerm));
  }

  get total() {
    return this.filteredProducts
      .reduce((sum, p) => sum + p.price, 0);
  }

  ngOnInit() { this.loadProducts(); }

    this.loading = true;
    this.service.getProducts().subscribe(data => {
      this.products = data;
      this.loading  = false;
    });
  }
}

Dopo (con signal)

export class NewComponent {
  products   = signal<Product[]>([]);
  loading    = signal(false);
  searchTerm = signal('');

  filteredProducts = computed(() =>
    this.products()
      .filter(p => p.name.includes(this.searchTerm()))
  );

  total = computed(() =>
    this.filteredProducts()
      .reduce((sum, p) => sum + p.price, 0)

  ngOnInit() { this.loadProducts(); }

    this.loading.set(true);
    this.service.getProducts().subscribe(data => {
      this.products.set(data);
      this.loading.set(false);
    });
  }
}

Pattern corretto: Service con entrambi

@Injectable({ providedIn: 'root' })
export class ProductsService {
  private http = inject(HttpClient);

  // ===== STATO (Signal) - condiviso tra componenti =====
  private _products = signal<Product[]>([]);
  private _loading  = signal(false);
  private _error    = signal<string | null>(null);

  readonly products       = this._products.asReadonly();
  readonly loading        = this._loading.asReadonly();
  readonly error          = this._error.asReadonly();
  readonly totalProducts  = computed(() => this._products().length);
  readonly activeProducts = computed(() => this._products().filter(p => p.active));

  // ===== METODI HTTP (Observable) - operazioni async =====
  getProducts(): Observable<Product[]> {
    return this.http.get<Product[]>('/api/products');
  }

  createProduct(product: Omit<Product, 'id'>): Observable<Product> {
    return this.http.post<Product>('/api/products', product);
  }

  deleteProduct(id: number): Observable<void> {
    return this.http.delete<void>(`/api/products/${id}`);
  }

  // ===== METODI PUBBLICI - aggiornano lo stato =====
  loadProducts() {
    this._loading.set(true); this._error.set(null);
    this.getProducts().subscribe({
      next:  (data) => { this._products.set(data); this._loading.set(false); },
      error: (err)  => { this._error.set('Errore caricamento'); this._loading.set(false); }
    });
  }

  addProduct(product: Omit<Product, 'id'>) {
    this._loading.set(true);
    this.createProduct(product).subscribe({
      next:  (created) => { this._products.update(list => [...list, created]); this._loading.set(false); },
      error: (err)     => { this._error.set('Errore creazione'); this._loading.set(false); }
    });
  }

  removeProduct(id: number) {
    this._loading.set(true);
    this.deleteProduct(id).subscribe({
      next:  ()    => { this._products.update(list => list.filter(p => p.id !== id)); this._loading.set(false); },
      error: (err) => { this._error.set('Errore eliminazione'); this._loading.set(false); }
    });
  }
}

Component usa i Signal del Service 

export class ProductsComponent {
  service = inject(ProductsService);

  // Leggi signal dal service
  products      = this.service.products;
  loading       = this.service.loading;
  error         = this.service.error;
  totalProducts = this.service.totalProducts;

  // Signal locale + computed che usa signal del service
  searchTerm = signal('');
  filteredProducts = computed(() =>
    this.service.products().filter(p => p.name.includes(this.searchTerm()))
  );

  ngOnInit() { this.service.loadProducts(); }
  onAdd(product: Omit<Product, 'id'>) { this.service.addProduct(product); }
  onDelete(id: number)                  { this.service.removeProduct(id); }
}
Pattern Quando Esempi
Service con Signal Più component accedono agli stessi dati, single source of truth Cart, Auth, Notifications
Service solo Observable Ogni component gestisce il proprio stato, chiamate one-off API wrapper, Logger
Pattern ibrido Stato condiviso + ricerche/filtri indipendenti per component UserService (currentUser signal + searchUsers observable)
Regola finale
❌ getProducts(): Signal<Product[]> — non fare questo
✅ getProducts(): Observable<Product[]> — metodo HTTP ritorna sempre Observable
✅ private _products = signal<Product[]>([]) — stato condiviso come signal
✅ loadProducts() { this.getProducts().subscribe(data => this._products.set(data)) }
