    <!-- Header, Lista, Errore -->
  </Grid>
</ContentPage>

RefreshView + CollectionView con DataTemplate Tipatolista
<RefreshView Grid.Row="1"
             Command="{Binding CaricaBuoniCommand}"
             IsRefreshing="{Binding IsBusy}">

  <CollectionView ItemsSource="{Binding Buoni}"
                  SelectionMode="None"
                  EmptyView="Nessun buono assegnato.">

    <CollectionView.ItemTemplate>
      <DataTemplate x:DataType="m:BuonoTestata">  <!-- type-safe binding -->
        <Grid Padding="12,6">
          <Frame BackgroundColor="White" CornerRadius="8"
                 Padding="0" HasShadow="True">

            <Grid RowDefinitions="Auto,Auto,Auto" Padding="14,12">

              <!-- Riga 1: numero buono + badge stato -->
              <Grid Grid.Row="0" ColumnDefinitions="*,Auto">
                <Label Grid.Column="0" Text="{Binding TitoloDisplay}"
                       FontSize="15" FontAttributes="Bold"/>
                <Frame Grid.Column="1"
                       BackgroundColor="{Binding StatoColor,
                                        Converter={StaticResource HexColor}}"
                       CornerRadius="4" Padding="8,2" HasShadow="False">
                  <Label Text="{Binding StatoLabel}"
                         TextColor="White" FontSize="11" FontAttributes="Bold"/>
                </Frame>
              </Grid>

              <!-- Riga 2: cliente -->
              <Label Grid.Row="1" Text="{Binding RagSoc}"
                     FontSize="13" TextColor="#555555" Margin="0,2,0,6"/>

              <!-- Riga 3: progress bar -->
              <Grid Grid.Row="2" RowDefinitions="Auto,Auto">
                <Label Text="{Binding AvanzamentoLbl}"
                       FontSize="12" TextColor="#777"/>
                <ProgressBar Grid.Row="1"
                             Progress="{Binding Progresso,
                                        Converter={StaticResource Clamp}}"
                             ProgressColor="#0057A8"
                             Margin="0,4,0,0"/>
              </Grid>

            </Grid>

            <!-- TapGestureRecognizer con binding al ViewModel antenato -->
            <Frame.GestureRecognizers>
              <TapGestureRecognizer
                Command="{Binding Source={RelativeSource
                          AncestorType={x:Type vm:BuoniListViewModel}},
                          Path=ApriDettaglioCommand}"
                CommandParameter="{Binding .}"/>
            </Frame.GestureRecognizers>
          </Frame>
        </Grid>
      </DataTemplate>
    </CollectionView.ItemTemplate>
  </CollectionView>
</RefreshView>

TapGestureRecognizer con RelativeSource
Dentro un DataTemplate il BindingContext è il singolo item (BuonoTestata). Per accedere al ViewModel usa Source={RelativeSource AncestorType={x:Type vm:BuoniListViewModel}}. Poi CommandParameter="{Binding .}" passa l'item corrente come parametro.
