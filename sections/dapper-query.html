Dapper vs EF Core
Dapper è un micro-ORM: scrive SQL a mano, mappa automaticamente i risultati sul DTO. Ottimo per query complesse con JOIN multipli dove EF Core diventerebbe verboso. Non gestisce migrazioni o change tracking.
Query<T> — Versione Sincronasync
public List<OrderQuantoDto> SelectOrderQuanto(
    string annoEsercizio = "2025",
    int avanzMin = 10,
    int avanzMax = 90,
    string terminale = "TER01")
{
    var connection = _sqlConnection.CreateConnectionQuantoDb();
    if (connection == null)
        throw new InvalidOperationException("Impossibile ottenere la connessione.");

    var sql = @"
    SELECT QA_T_T_DOCP.Docume,
           Quanto.IS_T_R_DOCP.N_ID_R_DOCP,
           Quanto.IS_T_R_DOCP.C_ID_ARTICOLO,
           Quanto.IS_T_R_DOCP.C_ID_VAR_ART,
           Quanto.IS_T_R_DOCP.C_DES_R_DOCP,
           Quanto.IS_T_R_DOCP.N_QTA_USR_R_DOCP,
           Quanto.IS_T_R_DOCP.C_ID_UM
    FROM QA_T_T_DOCP
    INNER JOIN Quanto.IS_T_R_DOCP
        ON QA_T_T_DOCP.N_ID_DITTA = Quanto.IS_T_R_DOCP.N_ID_DITTA
       AND QA_T_T_DOCP.N_NUM_DOCP = Quanto.IS_T_R_DOCP.N_NUM_DOCP
       AND Quanto.IS_T_R_DOCP.N_ID_R_DOCP_SUP IS NULL
    WHERE QA_T_T_DOCP.C_ANNO_ESERCIZIO >= @AnnoEsercizio
      AND QA_T_T_DOCP.Avanz >= @AvanzMin
      AND QA_T_T_DOCP.Avanz < @AvanzMax
      AND QA_T_T_DOCP.Termin = @Terminale";

    try
    {
        using (connection)
        {
            var results = connection.Query<OrderQuantoDto>(
                sql,
                new {
                    AnnoEsercizio = annoEsercizio,
                    AvanzMin = avanzMin,
                    AvanzMax = avanzMax,
                    Terminale = terminale
                }
            ).ToList();

            return results;
        }
    }
    catch (Exception ex)
    {
        throw new InvalidOperationException(
            $"Errore recupero ordini: {ex.Message}", ex);
    }
}
QueryAsync<T> — Versione Asincronaasync
public async Task<List<OrderQuantoDto>> SelectOrderQuantoAsync(
    string annoEsercizio = "2025",
    int avanzMin = 10,
    int avanzMax = 90,
    string terminale = "TER01")
{
    var connection = _sqlConnection.CreateConnectionQuantoDb();
    if (connection == null)
        throw new InvalidOperationException("Impossibile ottenere la connessione.");

    // stessa variabile sql della versione sincrona...

    try
    {
        using (connection)
        {
            var results = await connection.QueryAsync<OrderQuantoDto>(
                sql,
                new {
                    AnnoEsercizio = annoEsercizio,
                    AvanzMin = avanzMin,
                    AvanzMax = avanzMax,
                    Terminale = terminale
                }
            );

            return results.ToList();
        }
    }
    catch (Exception ex)
    {
        throw new InvalidOperationException(
            $"Errore recupero ordini: {ex.Message}", ex);
    }
}
Differenza chiave
Query<T> restituisce IEnumerable<T> direttamente. QueryAsync<T> restituisce Task<IEnumerable<T>> — aggiungi .ToList() dopo await per materializzare.
