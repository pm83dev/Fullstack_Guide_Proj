Setup Progetto da Zero
Angular standalone + NGXS + HttpClient — dal nulla al primo dispatch
1 — Crea il progetto Angular
# Installa Angular CLI se non presente
npm install -g @angular/cli

# Crea il progetto (standalone di default da Angular 17+)
ng new my-app --standalone --routing --style=scss
cd my-app
ng serve
2 — Installa NGXS
npm install @ngxs/store @ngxs/devtools-plugin @ngxs/logger-plugin
3 — Configura app.config.ts
import { ApplicationConfig } from '@angular/core';
import { provideRouter } from '@angular/router';
import { provideStore } from '@ngxs/store';
import { provideNgxsDevtoolsPlugin } from '@ngxs/devtools-plugin';
import { provideHttpClient } from '@angular/common/http';
import { routes } from './app.routes';
import { UserState } from './state/user.state';

export const appConfig: ApplicationConfig = {
  providers: [
    provideRouter(routes),
    provideStore([UserState]),     // aggiungi i tuoi State qui
    provideNgxsDevtoolsPlugin(),   // Redux DevTools nel browser
    provideHttpClient()
  ]
};
4 — Struttura cartelle consigliata
src/app/
  app.config.ts
  app.routes.ts
  store/
    plant/
      plant.actions.ts   // tutte le Action
      plant.state.ts     // State + Selectors + @Action
      plant.model.ts     // interfacce StateModel + domain models
    alarms/
      alarms.actions.ts
      alarms.state.ts
  services/
    plant-api.service.ts // solo HTTP, niente NGXS
  components/
    plant-list/
5 — Registra State e Routing
// app.routes.ts
export const routes: Routes = [
  { path: '', redirectTo: 'users', pathMatch: 'full' },
  { path: 'users', component: UserListComponent },
  { path: 'users/:id', component: UserDetailComponent },
  { path: '**', redirectTo: 'users' }
];

// Per aggiungere un nuovo State → basta aggiungerlo in provideStore([...])
