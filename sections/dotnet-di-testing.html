Pattern — InjectApplicationServices Extensiondependency injection

Raggruppa tutte le registrazioni DI del layer Application in un unico metodo estensione statico, mantenendo Program.cs pulito.

// Application/InjectAppServices.cs
using Microsoft.Extensions.DependencyInjection;
using Application.Services;
using Application.Commands;

namespace Application
{
    public static class InjectAppServices
    {
        public static IServiceCollection InjectApplicationServices(
            this IServiceCollection self)
        {
            self.AddScoped<IAccountService, AccountService>();
            self.AddScoped<IDeviceDataService, DeviceDataService>();
            self.AddScoped<DppMachineService>();
            self.AddAutoMapper(typeof(MappingProfile));
            self.AddHostedService<MqttBackgroundService>();
            return self;
        }
    }
}
// Program.cs — uso
builder.Services.InjectApplicationServices();
Vantaggi del pattern
Separa la responsabilità di registrazione per layer (Application, Infrastructure, ecc.). Program.cs resta leggibile. Ogni layer può avere il proprio extension method. Facilita il testing perché si possono sovrascrivere singoli servizi.
xUnit — Creazione Progetto di Testxunit
# 1. Crea il progetto di test
dotnet new xunit -n MyApp.Test

# 2. Aggiungi i pacchetti necessari
dotnet add package xunit
dotnet add package Moq
dotnet add package Microsoft.NET.Test.Sdk
dotnet add package xunit.runner.visualstudio
dotnet add package coverlet.collector

# 3. Aggiungi riferimento al progetto da testare
dotnet add reference ../MyApp/MyApp.csproj

# 4. Aggiungi il progetto test alla solution
dotnet sln add MyApp.Test/MyApp.Test.csproj

# 5. Esegui i test
cd MyApp.Test
dotnet test

Moq — Pattern Base per Test con Mockmoq
using Moq;
using Xunit;

public class AccountServiceTests
{
    private readonly Mock<IAccountRepository> _repoMock;
    private readonly AccountService _sut; // System Under Test

    public AccountServiceTests()
    {
        _repoMock = new Mock<IAccountRepository>();
        _sut = new AccountService(_repoMock.Object);
    }

    [Fact]
    public async Task GetById_ReturnsUser_WhenExists()
    {
        // Arrange
        var expected = new User { Id = 1, Name = "Mario" };
        _repoMock.Setup(r => r.GetByIdAsync(1))
                 .ReturnsAsync(expected);

        // Act
        var result = await _sut.GetByIdAsync(1);

        // Assert
        Assert.Equal(expected.Name, result.Name);
        _repoMock.Verify(r => r.GetByIdAsync(1), Times.Once);
    }
}
