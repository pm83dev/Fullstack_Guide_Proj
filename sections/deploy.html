Deploy VPS — Docker + Nginx + SSL
Aggiungere una nuova app .NET+Angular su VPS esistente con Docker Compose
Infrastruttura di riferimento
ComponenteDettaglio
Compose filedocker-compose-ssl.yml (sempre con -f)
Reverse proxyNginx (container nginx-proxy)
SSLCertbot + Let's Encrypt (rinnovo auto ogni 12h)
DatabaseSQL Server 2022 (container sqlserver, condiviso)
Rete Dockergadis-network (external)
Step 1 — Cartelle volumi su host
sudo mkdir -p /srv/[nome-app]/images
sudo mkdir -p /srv/[nome-app]/documents
sudo chown -R deployuser:deployuser /srv/[nome-app]
# ⚠️ Sempre chown PRIMA del trasferimento file con WinSCP
Step 2 — Blocco servizio in docker-compose-ssl.yml
[nome-servizio]:
  image: pm83dev/[nome-immagine]:latest
  container_name: [nome-servizio]
  restart: unless-stopped
  environment:
    - DOTNET_ENVIRONMENT=Production
    - ASPNETCORE_URLS=http://+:5000
    - ConnectionStrings__Fashion=Server=sqlserver,1433;Database=[DB];...
  depends_on:
    - sqlserver
  networks:
    - gadis-network
  expose:
    - "5000"           # solo interno, Nginx è l'unico punto pubblico
  volumes:
    - /srv/[nome-app]/images:/data/images
    - /srv/[nome-app]/documents:/data/documents
# ⚠️ Indentazione YAML: sempre 2 spazi, MAI tab
# ⚠️ ConnectionStrings__ (doppio underscore) = : nella gerarchia JSON
Step 3 — Restore DB in SQL Server container
# Verifica nomi logici del backup
docker exec -it sqlserver /opt/mssql-tools18/bin/sqlcmd \
  -S localhost -U sa -P "1.Password!" -C \
  -Q "RESTORE FILELISTONLY FROM DISK = '/backups/[file.back]'"

# Restore con MOVE (obbligatorio per path Windows → Linux)
docker exec -it sqlserver /opt/mssql-tools18/bin/sqlcmd \
  -S localhost -U sa -P "1.Password!" -C \
  -Q "RESTORE DATABASE [NomeDB] FROM DISK = '/backups/[file.back]' \
  WITH MOVE '[LogicalName_dati]' TO '/var/opt/mssql/data/[NomeDB].mdf', \
  MOVE '[LogicalName_log]' TO '/var/opt/mssql/data/[NomeDB]_log.ldf', REPLACE"
Step 4-6 — Nginx + SSL
# 4. Crea conf HTTP per webroot challenge Certbot
nano /home/deployuser/nginx/conf.d/[nome-app].conf
docker exec nginx-proxy nginx -s reload

# 5. Genera certificato SSL
docker exec certbot certbot certonly --webroot \
  -w /var/www/certbot -d [sottodominio.dominio.com] \
  --email [email] --agree-tos --no-eff-email

# 6. Aggiungi blocco HTTPS alla conf Nginx
# proxy_pass http://[nome-servizio]:5000  ← nome servizio Docker, non container_name
docker exec nginx-proxy nginx -s reload
Step 7-9 — Deploy e verifica
# Pull immagine e avvio (senza toccare gli altri container)
docker compose -f docker-compose-ssl.yml pull [nome-servizio]
docker compose -f docker-compose-ssl.yml up -d [nome-servizio]
docker exec nginx-proxy nginx -s reload

# Verifica
docker ps | grep [nome-servizio]
docker logs [nome-servizio] --tail 50
docker exec -it [nome-servizio] ls -la /data/images
