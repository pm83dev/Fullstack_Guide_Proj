// user.actions.ts — pattern namespace
export namespace UserActions {

  export interface LoadUsersPayload {
    page: number;
    pageSize: number;
    filters?: {
      name?: string;
      isActive?: boolean;
    };
  }

  export class LoadUsers {
    static readonly type = '[User List] Load Users';
    constructor(public payload: LoadUsersPayload) {}
  }

  export class CreateUser {
    static readonly type = '[User Form] Create User';
    constructor(public payload: CreateUserDto) {}
  }

  export class UpdateUser {
    static readonly type = '[User Form] Update User';
    constructor(public id: string, public payload: UpdateUserDto) {}
  }

  export class LoadUsersFailure {
    static readonly type = '[User API] Load Users Failure';
    constructor(public payload: { error: any; statusCode: number }) {}
  }
}

// Uso nel componente
this.store.dispatch(new UserActions.LoadUsers({
  page: 1, pageSize: 10, filters: { isActive: true }
}));
Dispatch Condizionale e Multiplocomponent
async saveUser(userData: any) {
  // Dispatch condizionale: create o update
  if (userData.id) {
    await this.store.dispatch(
      new UserActions.UpdateUser(userData.id, {
        name: userData.name, email: userData.email
      })
    ).toPromise();
  } else {
    await this.store.dispatch(
      new UserActions.CreateUser({
        name: userData.name, email: userData.email, password: userData.password
      })
    ).toPromise();
  }

  // Dispatch multiplo in un colpo solo
  this.store.dispatch([
    new UserActions.LoadUsers({ page: 1, pageSize: 10 }),
    new NotificationActions.ShowSuccess('Salvato!')
  ]);
}
Validazione del Payload nello Statestate
@Action(UserActions.CreateUser)
createUser(ctx: StateContext<UserStateModel>, action: UserActions.CreateUser) {
  const { name, email, password } = action.payload;

  // Validazione prima di procedere
  if (!name?.trim()) {
    return ctx.dispatch(new UserActions.CreateUserFailure({
      error: 'Name is required', field: 'name'
    }));
  }
  if (!isValidEmail(email)) {
    return ctx.dispatch(new UserActions.CreateUserFailure({
      error: 'Invalid email', field: 'email'
    }));
  }

  ctx.patchState({ loading: true });
  return this.userService.createUser(action.payload).pipe(
    tap(user => ctx.dispatch(new UserActions.CreateUserSuccess(user))),
    catchError(err => of(ctx.dispatch(
      new UserActions.CreateUserFailure({ error: err.message, statusCode: err.status })
    )))
  );
}
Destrutturazione — Payload con Più Campistate
@Action(UserActions.LoadUsers)
loadUsers(ctx: StateContext<UserStateModel>, action: UserActions.LoadUsers) {
  // Destrutturazione per accesso pulito
  const { page, pageSize, filters } = action.payload;

  ctx.patchState({ loading: true, error: null });

  return this.userService.getUsers(page, pageSize, filters).pipe(
    tap(result => {
      ctx.dispatch(new UserActions.LoadUsersSuccess(result.items));
    }),
    catchError(error => of(
      ctx.dispatch(new UserActions.LoadUsersFailure({
        error: error.message, statusCode: error.status
      }))
    ))
  );
}

// Payload doppio: id + dati separati
@Action(UserActions.UpdateUser)
updateUser(ctx: StateContext<UserStateModel>, action: UserActions.UpdateUser) {
  const userId    = action.id;       // string
  const updateData = action.payload; // UpdateUserDto

  return this.userService.updateUser(userId, updateData).pipe(
    tap(updatedUser => {
      const state = ctx.getState();
      ctx.patchState({
        users: state.users.map(u => u.id === updatedUser.id ? updatedUser : u),
        loading: false
      });
    })
  );
}
Spread Operator — Aggiungere Campi al Payload nel Serviceservice

Pattern tipico: il component passa un filtro, il service aggiunge un campo preso dallo store (es. AnaCod dall'utente autenticato).

getDocuments(payload?: Params): Observable<IDocumentModel> {
  const user = this.store.selectSnapshot(AccountState.user);
  const tmpAnaCod = user?.id ?? 0;
  const url = `${environment.API_URL}/Documenti/getAllDocs`;

  // Spread: copia il payload + aggiunge AnaCod
  const params = { ...payload, AnaCod: tmpAnaCod };
  // { OrdNum: 7 } + { AnaCod: 93 } → { OrdNum: 7, AnaCod: 93 }

  return this.http.get<IDocumentModel>(url, { params });
  // Angular genera: ?OrdNum=7&AnaCod=93  ✓
}
⚠ Errore comune — wrapping doppio
SBAGLIATO: getDocuments({ payload }) → il service riceve { payload: {...} }
CORRETTO: getDocuments(action.payload) → il service riceve { OrdNum: 7 }
