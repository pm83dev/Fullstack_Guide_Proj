Dispatcha azioni. Legge lo stato con @Select o selectSnapshot. Il template usa async pipe.
product.component.ts component 
@Component({
  selector: 'app-product',
  template: `
    <div *ngFor="let product of products$ | async">
      {{ product.name }}
    </div>
  `,
})
export class ProductComponent implements OnInit {
  // Lettura reattiva — si aggiorna automaticamente
  @Select(ProductState.products) products$!: Observable<IProduct[]>;

  constructor(private store: Store) {}

  ngOnInit(): void {
    // Dispatch con payload opzionale
    this.store.dispatch(new GetProductsAction({ search: 'shirt', sortBy: 'asc' }));
  }
}

// selectSnapshot = lettura sincrona one-shot (es: ngOnInit, guard)
const user = this.store.selectSnapshot(AuthState.user);
