4 — Model, Actions e Statestate
// models/product.model.ts
export interface Product {
  id: number; name: string; price: number; description: string;
}
export interface ProductStateModel {
  products: Product[]; loading: boolean; error: any;
}

// store/product.actions.ts
export class LoadProducts {
  static readonly type = '[Products] Load';
}
export class LoadProductsSuccess {
  static readonly type = '[Products] Load Success';
  constructor(public products: Product[]) {}
}
export class LoadProductsFail {
  static readonly type = '[Products] Load Fail';
  constructor(public error: any) {}
}

// store/product.state.ts
@State<ProductStateModel>({
  name: 'products',
  defaults: { products: [], loading: false, error: null }
})
@Injectable()
export class ProductState {
  constructor(private svc: ProductService) {}

  @Selector() static products(s: ProductStateModel) { return s.products; }
  @Selector() static loading(s: ProductStateModel) { return s.loading; }

  @Action(LoadProducts)
  load(ctx: StateContext<ProductStateModel>) {
    ctx.patchState({ loading: true, error: null });
    return this.svc.getProducts().pipe(
      tap(p => ctx.dispatch(new LoadProductsSuccess(p))),
      catchError(e => of(ctx.dispatch(new LoadProductsFail(e))))
    );
  }

  @Action(LoadProductsSuccess)
  success(ctx: StateContext<ProductStateModel>, a: LoadProductsSuccess) {
    ctx.patchState({ products: a.products, loading: false });
  }

  @Action(LoadProductsFail)
  fail(ctx: StateContext<ProductStateModel>, a: LoadProductsFail) {
    ctx.patchState({ error: a.error, loading: false });
  }
}
5 — Componenti Standalone + Routingcomponent
// Lista prodotti
@Component({
  standalone: true,
  imports: [CommonModule, RouterModule],
  template: `
    <div *ngIf="loading$ | async">Caricamento...</div>
    <ul *ngIf="(products$ | async) as products">
      <li *ngFor="let p of products">
        <a [routerLink]="['/products', p.id]">
          {{ p.name }} — {{ p.price | currency }}
        </a>
      </li>
    </ul>
  `
})
export class ProductListComponent implements OnInit {
  products$ = this.store.select(ProductState.products);
  loading$  = this.store.select(ProductState.loading);

  constructor(private store: Store) {}
  ngOnInit() { this.store.dispatch(new LoadProducts()); }
}

// Dettaglio prodotto — legge id dalla rotta + filtra dallo store
@Component({ standalone: true, imports: [CommonModule, RouterModule], template: `...` })
export class ProductDetailComponent implements OnInit {
  product$: Observable<Product | undefined>;

  constructor(private route: ActivatedRoute, private store: Store) {}

  ngOnInit() {
    this.product$ = this.route.paramMap.pipe(
      switchMap(params => {
        const id = Number(params.get('id'));
        return this.store.select(ProductState.products).pipe(
          map(products => products.find(p => p.id === id))
        );
      })
    );
  }
}

// app.routes.ts
export const routes: Routes = [
  { path: '', redirectTo: '/products', pathMatch: 'full' },
  { path: 'products', component: ProductListComponent },
  { path: 'products/:id', component: ProductDetailComponent },
  { path: '**', redirectTo: '/products' }
];

7 — Servire Angular da .NET (Produzione)deploy
# Build Angular
ng build --configuration production
# Output in: dist/angular-ngxs-demo/
# Copiare il contenuto in: wwwroot/

// Program.cs — serve Angular + gestisce routing SPA
app.UseDefaultFiles();    // cerca index.html nella root
app.UseStaticFiles();    // serve file da wwwroot
app.MapControllers();
app.MapFallbackToFile("index.html");
// Qualsiasi path non API (es. /products/5) → index.html → Angular gestisce il routing
Ordine middleware importante
Le rotte API devono essere prima del MapFallbackToFile. Se invertito, ogni richiesta API restituisce index.html invece del JSON.
