Mantiene lo stato centrale. Gestisce le Action con @Action. Espone dati con @Selector.
product.state.ts — struttura completa state 
import { State, Action, StateContext, Selector } from '@ngxs/store';

export interface ProductStateModel {
  products: IProduct[];
  selectedProduct: IProduct | null;
}

@State<ProductStateModel>({
  name: 'product',
  defaults: { products: [], selectedProduct: null },
})
@Injectable()
export class ProductState {
  constructor(private productService: ProductService) {}

  // 🔹 SELECTOR
  @Selector()
  static products(state: ProductStateModel) {
    return state.products;
  }

  // 🔹 ACTION HANDLER
  @Action(GetProductsAction)
  getProducts(ctx: StateContext<ProductStateModel>, action: GetProductsAction) {
    return this.productService.getProducts(action.payload).pipe(
      tap((result) => {
        ctx.patchState({ products: result.data || [] });
      }),
    );
  }
}
⚠️ Errore comune
Il tipo generico di @State<T> e StateContext<T> deve essere lo State Model, non il Domain Model né la risposta API.
