State per Array di Oggetti
La distinzione fondamentale: Domain Model ≠ State Model
data.interface.ts — separazione corretta state 
// Domain model — rappresenta UN oggetto dell'API
export interface DataItem {
  id: number;
  name: string;
  value: string;
}

// State model — forma dell'intero snapshot NGXS
export interface DataStateModel {
  items: DataItem[];  // l'array come proprietà
  loading: boolean;
  error: string;
}
data.state.ts — pattern corretto state 
@State<DataStateModel>({   // <-- DataStateModel, non DataItem!
  name: 'data',
  defaults: { items: [], loading: false, error: '' },
})
export class DataState {

  @Selector()
  static items(state: DataStateModel): DataItem[] {
    return state.items;  // ritorna l'array direttamente
  }

  @Action(LoadApiData)
  loadApiData(ctx: StateContext<DataStateModel>) {
    ctx.patchState({ loading: true, error: '' });
    return this.apiService.getData().pipe(
      tap((response) => {
        ctx.patchState({ items: response.items, loading: false });
      }),
      catchError((error) => {
        ctx.patchState({ loading: false, error: 'Errore caricamento' });
        return throwError(() => error);
      }),
    );
  }
}
